Source Code (ST)

(*	Start assignment *)	
	UniqueID := IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_RANGE_START;
	
	(*	Start assignment > Expansion *)
	UniqueID := UniqueID + (IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_RANGE_SIZE * ExpID);
	
	CASE IOPinType OF
	
	Enum_IO_PinType#IOPinType_AI:	
		(*	Start assignment > Expansion > Section AI *)
		UniqueID := UniqueID + IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_AI_OFFSET;
		
	Enum_IO_PinType#IOPinType_DI:
		(*	Start assignment > Expansion > Section DI *)
		UniqueID := UniqueID + IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_DI_OFFSET;
	
	Enum_IO_PinType#IOPinType_AO:
		(*	Start assignment > Expansion > Section AO *)
		UniqueID := UniqueID + IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_AO_OFFSET;
		
	Enum_IO_PinType#IOPinType_DO:
		(*	Start assignment > Expansion > Section DO *)
		UniqueID := UniqueID + IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_DO_OFFSET;
		
	Enum_IO_PinType#IOPinType_SMV:
		(*	Start assignment > Expansion > Section SMV *)
		UniqueID := UniqueID + IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_SMV_OFFSET;
		
	END_CASE;

	(*	Start assignment > Expansion > Section SMV > Pin *)
	UniqueID := UniqueID + (IOPinTypeIndex * IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_FIELDS);
	
	(*	Start assignment > Expansion > Section SMV > Pin > Field *)
	UniqueID := UniqueID + TO_UDINT(IOFieldIndex);

	GetExpansionVoiceUniqueId := UniqueID;
