	(*	Email queue not available	*)
	eLogStatus := Log_Status_t#Log_Disabled;
	eLogError := Log_Error_t#LogSuccess;
	xInit := FALSE;

{ IFDEF: clsQueueEmailToSend}
(*----------------------------------*)
(*		INITIALIZE THE EMAIL QUEUE	*)
(*----------------------------------*)
	IF (xInit = FALSE) THEN
	
		xInit := TRUE;

		(*	"Create" the queue	*)		
		xQueueEnabled := clsQueueEmailToSend.Create(usiQueueMaxNumEvent,			(*	usiItemsMax		*)
												 TO_UINT(SIZEOF(eventLogData)),	(*	uiItemSize		*)
		                                         pntQueueBuff,					(*	pntDataBlock	*)
		                                         uiQueueBuffSize);				(*	uiDataBlockSize	*)
		                                         
		
		IF (xQueueEnabled = TRUE) THEN
			(*	Queue ready		*)
			eEmailStatus := Email_Status_t#Ready;		
		ELSE
			(* Issue on the queue initialization	*)
			eEmailStatus := Email_Status_t#Error_Config;		
		END_IF;
		
		usiQueueMaxItems  := 0;
				
		
	END_IF;


	(*	Configuration error ==> Exit	*)
	IF (eEmailStatus = Email_Status_t#Error_Config) THEN

		RETURN;
		
	END_IF;
	



(*--------------------------*)
(*		MODE MANAGEMENT		*)
(*--------------------------*)

	IF (xEnable = TRUE) THEN
		(*	Email enabled		*)
		
		IF (xStop = TRUE) THEN
			(*	Email stopped	*)
			eEmailMode := Email_Mode_t#Stop;
		ELSE
			(*	Otherwise Email is running	*)
			eEmailMode := Email_Mode_t#Running;
		END_IF;
		
	ELSE
		(*	Email disabled	*)
		eEmailMode := Email_Mode_t#Disabled;
	END_IF;


	

	(*------------------------------*)	
	(*	DEQUEUE ALL THE EVENTS		*)
	(*------------------------------*)			

	IF eEmailMode = Email_Mode_t#Running THEN

		xExit := TRUE;
		
		usiAux := 0;
		
		WHILE ((clsQueueEmailToSend.Empty() = FALSE) AND (xExit = TRUE)) DO
		
			xDummy := clsQueueEmailToSend.Dequeue(ADR(eventLogData));
		
			IF (xDummy = TRUE) THEN
				
				usiAux := usiAux + 1;
				
				(*	Send E-Mail	*)
	
				sysLogWriteMsg(0, 'EventUniqueID: %d', eventLogData.EventUniqueID, 0, 0, 0);	
									
				fbEventSendNotification.event := eventLogData;
				fbEventSendNotification();					

				IF fbEventSendNotification.NotificationStatus > 0 THEN													

					eEmailStatus := Email_Status_t#Error_Send_Email;		

				END_IF;				
										  			
			END_IF;
			
		END_WHILE;
		
		
		(*	Refresh the max number of items found into the queue	*)
		IF (usiAux > usiQueueMaxItems) THEN		
			usiQueueMaxItems := usiAux;
		END_IF;
	
	END_IF;	
	
		

{ ENDIF }
