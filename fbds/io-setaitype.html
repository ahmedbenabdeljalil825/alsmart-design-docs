IO_SetAIType

Source Code (ST)

CASE uiAIDIType OF
		1:	(*	USED AS AI	*)

			CASE usiAIType OF
		
				Enum_AI_Type#AI_0_10_V :
				
					IoTypeValue := IoType_t#IoType_0_10V;
		
				Enum_AI_Type#AI_0_1_V :
				
					IoTypeValue := IoType_t#IoType_0_1V;
		
				Enum_AI_Type#AI_0_20_mA :
				
					IoTypeValue := IoType_t#IoType_0_20mA;
		
				Enum_AI_Type#AI_0_5_V :
				
					IoTypeValue := IoType_t#IoType_0_5V;
		
				Enum_AI_Type#AI_4_20_mA :
				
					IoTypeValue := IoType_t#IoType_4_20mA;
		
				Enum_AI_Type#AI_PT1000,
				Enum_AI_Type#AI_PT1000_TABLE:
				
					IoTypeValue := IoType_t#IoType_Pt1000;
		
				Enum_AI_Type#AI_PTC1000 :
				
					IoTypeValue := IoType_t#IoType_Ptc1000;
		
				Enum_AI_Type#AI_AKS32 :

					IoTypeValue := IoType_t#IoType_0_5V_Ratio;
					
				Enum_AI_Type#AI_RATIOMETRIC :

					IoTypeValue := IoType_t#IoType_0_5V_Ratio;
					
				Enum_AI_Type#AI_NI1000TK5:	
																				
					IoTypeValue := IoType_t#IoType_Pt1000Hr;

				Enum_AI_Type#AI_NTC_100K,
				Enum_AI_Type#AI_NTC_10K,
				Enum_AI_Type#AI_NTC_2K,
				Enum_AI_Type#AI_NTC_5K,
				Enum_AI_Type#AI_NTC_TABLE,
				Enum_AI_Type#AI_NTC_TABLE2:

					IoTypeValue := IoType_t#IoType_Ntc;

				ELSE
					IoTypeValue := IoType_t#IoType_Ntc;
		
			END_CASE;

		2:	(*	USED AS NORMAL DI	*)

			IoTypeValue := IoType_t#IoType_DigInput;

		6:	(*	USED AS DI PULSE COUNTER	*)

			IoTypeValue := IoType_t#IoType_PulseCounter;

	END_CASE;
				
	retVal := Ai_SetType(DeviceID,		(*	Device	*)
						 uiAIIndex, 	(*	Inputdx	*)
						 IoTypeValue);	(*	Type	*)	
		
	IO_SetAIType := retVal;
