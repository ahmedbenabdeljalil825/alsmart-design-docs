GetIOConfigParamUniqueID

I/O Management Intermediate FunctionBlock v1.0.0

Description

Fill UniqueIDs of configuration values for specified pin

Inputs (4)

Name Type Init Description
ExpID Device_t Device
IOPinType Enum_IO_PinType IO Pin Type
IOPinTypeIndex USINT Index of the pin associated with the IO type (IOType).
IOConfigParamUniqueID TIOManagerConfigParams^ TIOManagerConfigParams filled with result

Outputs (1)

Name Type Description
SwdcError SwdcError_t

Source Code (ST)

(*	Check device *)
	
	IF NOT (ExpID >= Device_t#MainController AND ExpID <= Device_t#Expansion20) THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	END_IF;
	
	(*	Check pin type *)
	
	IF NOT (IOPinType = Enum_IO_PinType#IOPinType_AI OR
			IOPinType = Enum_IO_PinType#IOPinType_DI OR
			IOPinType = Enum_IO_PinType#IOPinType_AO OR
			IOPinType = Enum_IO_PinType#IOPinType_DO OR
			IOPinType = Enum_IO_PinType#IOPinType_SMV) THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	END_IF;
	
	(*	Check pin index *)
	
	IF IOPinType = Enum_IO_PinType#IOPinType_AI AND IOPinTypeIndex >= IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_AI THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	ELSIF IOPinType = Enum_IO_PinType#IOPinType_DI AND IOPinTypeIndex >= IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_DI THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	ELSIF IOPinType = Enum_IO_PinType#IOPinType_AO AND IOPinTypeIndex >= IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_AO THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	ELSIF IOPinType = Enum_IO_PinType#IOPinType_DO AND IOPinTypeIndex >= IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_DO THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	ELSIF IOPinType = Enum_IO_PinType#IOPinType_SMV AND IOPinTypeIndex >= IOCONFIG_UNIQUEID_AUTOASSIGNMENT_EXPANSION_MAX_SMV THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;
	END_IF;
	
	IF IOConfigParamUniqueID = NULL THEN
		SwdcError := SwdcError_t#SwdcError_InvalidArgument;
		RETURN;	
	END_IF;
	
	(*	Fill object *)
	
	IOConfigParamUniqueID^.FunctionID := 			GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_FunctionID);
	IOConfigParamUniqueID^.TypePolarity :=			GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_TypePolarity);
	IOConfigParamUniqueID^.OverrideEnable := 		GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_OverrideEnable);
	IOConfigParamUniqueID^.Override := 				GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Override);
	
	IF IOPinType = Enum_IO_PinType#IOPinType_AI THEN
		IOConfigParamUniqueID^.Decimals := 				GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Decimals);
		IOConfigParamUniqueID^.Min := 					GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Min);
		IOConfigParamUniqueID^.Max := 					GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Max);
		IOConfigParamUniqueID^.LeftPercParamSet := 		GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_LeftPercParamSet);
		IOConfigParamUniqueID^.RightPercAdvancedSet := 	GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_RightPercAdvancedSet);
		IOConfigParamUniqueID^.Error := 				GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Error);
		IOConfigParamUniqueID^.FilterID := 				GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_FilterID);
		IOConfigParamUniqueID^.Calibration := 			GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Calibration);		
	ELSIF IOPinType = Enum_IO_PinType#IOPinType_AO THEN
		IOConfigParamUniqueID^.Decimals := 				GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_Decimals);
		IOConfigParamUniqueID^.Min := 					0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Max := 					0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.LeftPercParamSet := 		GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_LeftPercParamSet);
		IOConfigParamUniqueID^.RightPercAdvancedSet := 	GetExpansionVoiceUniqueId(ExpID, IOPinType, IOPinTypeIndex, Enum_IOConfigField#IOConfigField_RightPercAdvancedSet);
		IOConfigParamUniqueID^.Error := 				0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.FilterID := 				0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Calibration := 			0;	(* No configuration parameter *)
	ELSE	
		IOConfigParamUniqueID^.Decimals := 				0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Min := 					0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Max := 					0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.LeftPercParamSet := 		0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.RightPercAdvancedSet := 	0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Error := 				0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.FilterID := 				0;	(* No configuration parameter *)
		IOConfigParamUniqueID^.Calibration := 			0;	(* No configuration parameter *)
	END_IF;

	(* Done *)
	
	SwdcError := SwdcSuccess;
