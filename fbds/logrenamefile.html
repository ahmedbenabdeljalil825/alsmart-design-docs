(*	Select the Flash Memory	*)
	IF (xMemoryFlashPublic = TRUE) THEN
		(*	Public	*)
		memPeripheral := Peripheral_t#MemoryFlashStoragePublic;
	ELSE
		(*	Private	*)
		memPeripheral := Peripheral_t#MemoryFlashStorage;
	END_IF;		

	LogRenameFile := FALSE;


	(*	Save the file with the date	*)
	strDate := '';
	
	IF (xAddDate = TRUE) THEN
		
		strDate := DateYYYYMMDD(uiYear,
								usiMonth,
								usiDay);
	END_IF;
	
	usiCopyItem := 0;
	xCompleted := FALSE;					

					
	WHILE ((xCompleted = FALSE) AND (usiCopyItem < 10)) DO
			
		(*	Check if the "old file name" is available	*)
		strCopyItem := TO_STRING(usiCopyItem);
		strCopyFilePathAndName := strFilePathAndName;
		
		IF (xAddDate = TRUE) THEN
			strCopyFilePathAndName := CONCAT(strCopyFilePathAndName, strDate);
		END_IF;
		
		(*	Add the "Copy" index only if is not the first	*)
		IF (usiCopyItem > 0) THEN
			strCopyFilePathAndName := CONCAT(strCopyFilePathAndName, '_');
			strCopyFilePathAndName := CONCAT(strCopyFilePathAndName, strCopyItem);
		END_IF;
		
		strCopyFilePathAndName := CONCAT(strCopyFilePathAndName, strFileExtension);
												
		retSwdcError := FileSystem_OpenFile(memPeripheral,				(*	PeripheralID	*)
											strCopyFilePathAndName,		(*	FileName		*)
											'r',						(*	Mode			*)
											FALSE,						(*	DoubleCopy		*)
											?fileHandler				(*	FileHandler		*)
											);
		
		(*	This file name is available: copy the current log	*)				
		IF (retSwdcError <> SwdcError_t#SwdcSuccess) THEN
			
			strOldFilePathAndName := CONCAT(strFilePathAndName, strFileExtension);
							
			retSwdcError := FileSystem_CopyFile(memPeripheral,				(*	PeripheralID		*)
												strOldFilePathAndName,		(*	SourceFilename		*)
												memPeripheral,				(*	DestPeripheralID	*)
												strCopyFilePathAndName		(*	DestFilename		*)
												);																
							
			(*	Remove the current log file	*)
			IF (retSwdcError = SwdcError_t#SwdcSuccess) THEN
				retSwdcError := FileSystem_DeleteFile(memPeripheral,			(*	PeripheralID	*)
													  strOldFilePathAndName		(*	FileName		*)
													 );
								
				(*	Create a new one	*)									  
				IF (retSwdcError = SwdcError_t#SwdcSuccess) THEN
					LogRenameFile := TRUE;
				END_IF;															
			END_IF;
							
			xCompleted := TRUE;	(*	Copy saved	*)									
		END_IF;
						
		usiCopyItem := usiCopyItem + 1;
		
	END_WHILE;
