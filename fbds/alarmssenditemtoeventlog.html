AlarmsSendItemToEventLog

Alarms & Safety Basic Function v1.0.1

Description

Send the alarm activation/deactivation info to the EventLog file.

Return Value

BOOL

uiAlarmID UINT UINT, range [0, 65535]. You can find the Alarm ID List in Tab Project->Alarm Manager->Ungrouped_vars

Source Code (ST)

AlarmsSendItemToEventLog := FALSE;


{ IFDEF : clsQueueEventLog }
	(*	This code compiles only if it has included the standard EventLog library in the project	*)

	CurAlarmStatus := GetAlarmStatus(uiAlarmID);
	CurAlarmConfig := GetAlarmConfig(uiAlarmID);


	usiNumContextVars := CurAlarmConfig^.NumContextVars;
	PContextVar := CurAlarmConfig^.PContextVars;
	
	(*		Add item to Event Log		*)	
	FOR usiIndex := 0 TO 9 DO
	
		IF ((usiNumContextVars > 0) AND (usiIndex < usiNumContextVars)) THEN
			(*	Context's variable	*)
			VarContextaddr := @PContextVar.RefVar;
						
			CASE @PContextVar.VarType OF
			
			'BOOL':
				pBOOL := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pBOOL);
			
			'SINT':
				pSINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pSINT);
			
			'USINT',
			'BYTE':
				pUSINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pUSINT);

			'INT':
			
				pINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pINT);
			
			'UINT',
			'WORD':
				pUINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pUINT);

			'DINT':
			
				pDINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pDINT);

			'UDINT',
			'DWORD':
				pUDINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pUDINT);
			
			'LINT':
				pLINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pLINT);
			
			'ULINT',
			'LWORD':
				pULINT := VarContextaddr;
				eventLogData.AlarmContextVariablesValues[usiIndex] := TO_REAL(@pULINT);
			
			'REAL':
				eventLogData.AlarmContextVariablesValues[usiIndex] := @VarContextaddr;
			
			'LREAL':
				eventLogData.AlarmContextVariablesUniqueIDs[usiIndex] := 0;
				
			ELSE
			
				eventLogData.AlarmContextVariablesUniqueIDs[usiIndex] := 0;
			END_CASE;
			
			eventLogData.AlarmContextVariablesUniqueIDs[usiIndex] := @PContextVar.UniqueID;
		
			PContextVar := PContextVar + SIZEOF(@PContextVar);
		
		ELSE
			(*	Nothing				*)
			eventLogData.AlarmContextVariablesValues[usiIndex] := 0;
			eventLogData.AlarmContextVariablesUniqueIDs[usiIndex] := 0;		
		END_IF;
	
	END_FOR;

	eventLogData.EventTime := AlarmSystemActualRTCValue;
	eventLogData.EventType := uiEventType;
	eventLogData.AlarmContextVariablesNumber := usiNumContextVars;
	eventLogData.EventUniqueID := CurAlarmConfig^.UniqueId;		


	AlarmsSendItemToEventLog := clsQueueEventLog.Enqueue(ADR(eventLogData)); 

{ ENDIF }


{ IFDEF : clsQueueEmailToSend }
		
	retValEmailQueue := clsQueueEmailToSend.Enqueue(ADR(eventLogData)); 
	
{ ENDIF }
