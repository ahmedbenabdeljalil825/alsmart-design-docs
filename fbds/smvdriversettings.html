(* 
		Calculate the SMVDriver array index
	*)
				
	usiIndex := device * 2;
	IF peripheral = Peripheral_t#ComStepperDriver2 THEN
		usiIndex := usiIndex + 1;	
	END_IF;

	IF usiIndex > CONST_IOSMVLastIndex THEN

		RETURN;	
	
	END_IF;
	

	(* 
		Chech inputs in range
	*)
	IF uiNeutralZone > 100 THEN
		uiNeutralZone := 100;
	ELSIF uiNeutralZone < 0 THEN
		uiNeutralZone := 0;
	END_IF;

	IF usiValveODOnPowerFailurePerc > 100 THEN
		usiValveODOnPowerFailurePerc := 100;
	ELSIF usiValveODOnPowerFailurePerc < 0 THEN
		usiValveODOnPowerFailurePerc := 0;
	END_IF;		

	IF usiEmergencyClosingSpeedPerc > 200 THEN
		usiEmergencyClosingSpeedPerc := 200;
	ELSIF usiEmergencyClosingSpeedPerc < 50 THEN
		usiEmergencyClosingSpeedPerc := 50;
	END_IF;	
		
	IF usiCurrentBoostPerc > 150 THEN
		usiCurrentBoostPerc := 150;
	ELSIF usiCurrentBoostPerc < 100 THEN
		usiCurrentBoostPerc := 100;
	END_IF;
			
	IF usiCurrentBoostOdStopPerc > 100 THEN
		usiCurrentBoostOdStopPerc := 100;
	ELSIF usiCurrentBoostOdStopPerc < 0 THEN
		usiCurrentBoostOdStopPerc := 0;
	END_IF;	
		
	IF usiCurrentBoostMaxTime < 0 THEN
		usiCurrentBoostMaxTime := 0;
	END_IF;	
		
	IF usiThermalProtectionPerc > 100 THEN
		usiThermalProtectionPerc := 100;
	ELSIF usiThermalProtectionPerc < 0 THEN
		usiThermalProtectionPerc := 0;
	END_IF;		

	IF usiODThresholdForAutomaticOverdrive > 100 THEN
		usiODThresholdForAutomaticOverdrive := 100;			
	ELSIF usiODThresholdForAutomaticOverdrive < 0 THEN
		usiODThresholdForAutomaticOverdrive	:= 0;
	END_IF;

	IF uiTimeThresholdForAutomaticOverdrive > 3600 THEN
		uiTimeThresholdForAutomaticOverdrive := 3600;			
	ELSIF uiTimeThresholdForAutomaticOverdrive < 0 THEN
		uiTimeThresholdForAutomaticOverdrive := 0;
	END_IF;



	(*
		Check if there are any differences
	*)	
	IF uiLastNeutralZone = uiNeutralZone AND	
	   	usiValveODOnPowerFailurePerc = usiLastValveODOnPowerFailurePerc AND
	   	usiEmergencyClosingSpeedPerc = usiLastEmergencyClosingSpeedPerc AND
		usiCurrentBoostPerc = usiLastCurrentBoostPerc AND
		usiCurrentBoostOdStopPerc = usiLastCurrentBoostOdStopPerc AND
		usiCurrentBoostMaxTime = usiLastCurrentBoostMaxTime AND
		xOpenWireFaultEnable = xLastOpenWireFaultEnable AND
		usiThermalProtectionPerc = usiLastThermalProtectionPerc AND
		usiODThresholdForAutomaticOverdrive = usiLastODThresholdForAutomaticOverdrive AND
		uiTimeThresholdForAutomaticOverdrive = uiLastTimeThresholdForAutomaticOverdrive THEN

		RETURN;		
		
	END_IF;


	(*
		Assigns the new values	
	*)
		
	SMVDriver[usiIndex].uiNeutralZone := uiNeutralZone;
	SMVDriver[usiIndex].usiValveODOnPowerFailurePerc := usiValveODOnPowerFailurePerc;
	SMVDriver[usiIndex].usiEmergencyClosingSpeedPerc := usiEmergencyClosingSpeedPerc;
	SMVDriver[usiIndex].usiCurrentBoostPerc := usiCurrentBoostPerc;
	SMVDriver[usiIndex].usiCurrentBoostOdStopPerc := usiCurrentBoostOdStopPerc;
	SMVDriver[usiIndex].usiCurrentBoostMaxTime := usiCurrentBoostMaxTime;
	SMVDriver[usiIndex].xOpenWireFaultEnable := xOpenWireFaultEnable;
	SMVDriver[usiIndex].usiThermalProtectionPerc := usiThermalProtectionPerc;
	SMVDriver[usiIndex].uiODThresholdForAutomaticOverdrive := TO_UINT(usiODThresholdForAutomaticOverdrive) * 100;
	SMVDriver[usiIndex].uiTimeThresholdForAutomaticOverdrive := uiTimeThresholdForAutomaticOverdrive;

	(*
		Save the latest values	
	*)	
	uiLastNeutralZone := uiNeutralZone;
	usiLastValveODOnPowerFailurePerc := usiValveODOnPowerFailurePerc;
	usiLastEmergencyClosingSpeedPerc := usiEmergencyClosingSpeedPerc;
	usiLastCurrentBoostPerc := usiCurrentBoostPerc;
	usiLastCurrentBoostOdStopPerc := usiCurrentBoostOdStopPerc;
	usiLastCurrentBoostMaxTime := usiCurrentBoostMaxTime;
	xLastOpenWireFaultEnable := xOpenWireFaultEnable;
	usiLastThermalProtectionPerc := usiThermalProtectionPerc;	
	usiLastODThresholdForAutomaticOverdrive := usiODThresholdForAutomaticOverdrive;
	uiLastTimeThresholdForAutomaticOverdrive := uiTimeThresholdForAutomaticOverdrive;
