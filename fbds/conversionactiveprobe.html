ConversionActiveProbe

Inputs & Sensors Intermediate Function v1.0.29

Description

This function linearizes the active probes

REAL

Source Code (ST)

(*	Calculate the Left and Right row points boundary	*)
rLeftDelta := (rSourceLeftDeltaPercent * (rSourceMaximum - rSourceMinimum)) / 100.0;
rRightDelta := (rSourceRightDeltaPercent * (rSourceMaximum - rSourceMinimum)) / 100.0;


IF (rInValue <= rLeftDelta) THEN
	(*	Below the Left delta percentage	*)
	rVal := rDestMinimum;
	
ELSE
	IF (rInValue >= (rSourceMaximum - rRightDelta)) THEN
		(*	Above the Right delta percentage	*)
		rVal := rDestMaximum;
		
	ELSE
		(*	Inside the linearization range	*)
		rTemp := ((rSourceMaximum - rRightDelta) - (rSourceMinimum + rLeftDelta));

		IF (rTemp <> TO_REAL(0) ) THEN
			rVal := (rDestMaximum - rDestMinimum) / ((rSourceMaximum - rRightDelta) - (rSourceMinimum + rLeftDelta));
			rVal := rVal * (rInValue - (rSourceMinimum + rLeftDelta));
			rVal := rDestMinimum + rVal;
		ELSE
			(*	If rTemp = 0 something went wrong	*)
			rVal := 0;
		END_IF;
	
	END_IF;
END_IF;


ConversionActiveProbe := rVal;
