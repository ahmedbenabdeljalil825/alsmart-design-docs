GetEventLogItem

HMI & Display Intermediate FunctionBlock v1.0.0

Description

Extracts a single event from the Event Log file. NOTES: This function block must be used in the Backgroud task

Inputs (4)

Name Type Init Description
xMemoryFlashPublic BOOL BOOL, range [FALSE, TRUE]. TRUE: Memory Flash public FALSE: Memory Flash private
xFirstNext BOOL BOOL, range [FALSE, TRUE]. TRUE: return the newest item FALSE: return the previus one
pntDataBuff @USINT USINT. Memory block where to store the event
uiDataSize UINT UINT. Memory block size

Outputs (5)

Name Type Description
xLogEmpty BOOL BOOL, range [FALSE, TRUE]: TRUE: the event log file is empty
xLastRecord BOOL BOOL, range [FALSE, TRUE]: TRUE: last event
xError BOOL BOOL, range [FALSE, TRUE]: TRUE: error on loading the event
udiTime UDINT UDINT, unit [s]. Absolute time when the event was logged
uiTimeOffset UINT UINT, unit [s]. Offset time when the event was logged

xError := TRUE;
	xLastRecord := FALSE;


	IF (xFirstNext = TRUE) THEN
	
(*----------------------------------*)
(*		Set the search				*)	
(*----------------------------------*)

		xDummy := FALSE;
		xSearchEnabled := FALSE;
		
		xLogEmpty := TRUE;
		
		IF (clsEventLog.FileOpened() = FALSE) THEN

			strFile := CONCAT(STR_EVENTLOG_FILE_PATH, '/');
			strFile := CONCAT(strFile, STR_EVENTLOG_FILE_NAME);
			strFile := CONCAT(strFile, STR_EVENTLOG_FILE_EXTENSION);

			usiRet := clsEventLog.FilePresent(	xMemoryFlashPublic,	(*	xMemoryFlashPublic				*)
												strFile);			(*	strFilePathAndNameWithExtension	*)

			(*	usiRet = 0: Error on load the file	*)
			(*		   = 1: File not present		*)
			(*		   = 2 File present and ready	*)
			IF (usiRet = 2) THEN
				(*	File present and OK	*)
				xDummy := TRUE;
			END_IF;
		ELSE
			xDummy := TRUE;
		END_IF;
		
		IF (xDummy = TRUE) THEN
		
			clsEventLog.xHandlerActiveForSearch := TRUE;
			
			udiMaxNumRecords := clsEventLog.GetMaxNumRecords();
			udiNumRecordBuffer := clsEventLog.GetNumBufferRecords();
			
			xDummy := clsEventLog.GetLogInfo(ADR(eventLogInfo));
		
			IF (xDummy = TRUE) THEN
			
				udiRecordPosition := eventLogInfo.udiLastRecordOffset;
				
				IF ((udiRecordPosition >= udiMaxNumRecords) AND (eventLogInfo.udiFreeRecords >= udiMaxNumRecords)) THEN
					(*	Log Empty	*)
					
				ELSE 
					xLogEmpty := FALSE;
					xSearchEnabled := TRUE;
					udiRecordsCounter := 0;					
				END_IF;
			END_IF;
		END_IF;	
	END_IF;

(*----------------------------------*)
(*		Return the record			*)	
(*----------------------------------*)
	IF (xSearchEnabled = TRUE) THEN
	
		(*	Load record	*)
		xDummy := clsEventLog.GetRecord(udiRecordPosition,		(*	udiRecordOffset	*)
									    ADR(udiTime),			(*	pntTime			*)
									    ADR(uiTimeOffset),		(*	pntTimeOffset	*)
									    pntDataBuff,			(*	pntDataBuff		*)
									    ADR(uiDataSize));		(*	pntDataSize		*)
		
		udiRecordsCounter := udiRecordsCounter + 1;
		
		udiAux := udiNumRecordBuffer + udiMaxNumRecords;
		
		IF (xDummy = TRUE) THEN
			(*	Load record ok	*)
			xError := FALSE;
				
			(*	 Set the index for next record	*)				    
			IF (udiRecordPosition = 0) THEN
				(*	Now is pointing the record at the position 0	*)
				
				IF ((eventLogInfo.udiFreeRecords > 0) OR (udiRecordsCounter >= udiMaxNumRecords)) THEN
					(*	Last record		*)
					xLastRecord := TRUE;
					xSearchEnabled := FALSE;
				ELSE
					(*	Circular buffer	*)
					udiRecordPosition := udiAux - 1;				
				END_IF;		
			ELSE
				IF (udiRecordsCounter >= udiMaxNumRecords) THEN
					(*	Last record		*)
					xLastRecord := TRUE;
					xSearchEnabled := FALSE;
				ELSE
					(*	Circular buffer	*)
					udiRecordPosition := udiRecordPosition - 1;				
				END_IF;		
		
			END_IF;
		END_IF;
	END_IF;
