EventLog

HMI & Display Advanced FunctionBlock v1.0.0

Description

Event log that stores the resources present in the associated queue. uiQueueBuffSize >= (usiQueueMaxNumEvent * SIZEOF(TEventLog)) where: SIZEOF(TEventLog) = 92 BYTEs NOTES: This function block must be used in the Backgroud task

Inputs (9)

Name Type Init Description
xEnable BOOL BOOL, range [FALSE, TRUE].
xStop BOOL BOOL, range [FALSE, TRUE]. TRUE: stop the event log
udiNumRecords UDINT UDINT, range [1,100000]. Number of the log records
udiLogAppType UDINT UDINT. Appilcation code
uiLogSWVersion UINT UINT. Software version
usiQueueMaxNumEvent USINT USINT. Max number of the events that the queue can contain
pntQueueBuff @USINT USINT. Memory block allocated for the event queue
uiQueueBuffSize UINT UINT. Size of memory block allocated for the event queue
xMemoryFlashPublic BOOL BOOL, range [FALSE, TRUE]. TRUE: Memory Flash public FALSE: Memory Flash private

Outputs (3)

Name Type Description
eLogStatus Log_Status_t USINT. Log status
eLogError Log_Error_t USINT. Log error code
usiQueueMaxItems USINT USINT. Queue maximum number of events reached

{ IFDEF: NOT clsQueueEventLog}

	(*	Event queue not available	*)
	eLogStatus := Log_Status_t#Log_Disabled;
	eLogError := Log_Error_t#LogSuccess;
	xInit := FALSE;
	
	RETURN;
	
{ ENDIF }


{ IFDEF: clsQueueEventLog}
(*----------------------------------*)
(*		INITIALIZE THE EVENT LOG	*)
(*----------------------------------*)
	IF (xInit = FALSE) THEN
	
		xInit := TRUE;

		(*	"Create" the queue	*)		
		xQueueEnabled := clsQueueEventLog.Create(usiQueueMaxNumEvent,			(*	usiItemsMax		*)
												 TO_UINT(SIZEOF(eventLogData)),	(*	uiItemSize		*)
		                                         pntQueueBuff,					(*	pntDataBlock	*)
		                                         uiQueueBuffSize);				(*	uiDataBlockSize	*)
		                                         
		
		IF (xQueueEnabled = TRUE) THEN
			(*	Queue ready		*)
			eEventLogState := Log_State_t#Log_Disabled;		
		ELSE
			(* Issue on the queue initialization	*)
			eEventLogState := Log_State_t#Log_ErrorConfig;		
		END_IF;

		eLogStatus := Log_Status_t#Log_Disabled;
		eLogError := Log_Error_t#LogSuccess;
		
		(*	By default the event log is disabled	*)
		eEventLogMode_Old := Log_Mode_t#Disabled;
		
		usiQueueMaxItems  := 0;
				
		xDummy := clsEventLog.Init();
		
	END_IF;


	(*	Configuration error ==> Exit	*)
	IF (eEventLogState = Log_State_t#Log_ErrorConfig) THEN

		eLogStatus := Log_Status_t#Log_ConfigError;
		eLogError := Log_Error_t#LogError_QueueCreation;	
		RETURN;
	END_IF;
	

(*--------------------------*)
(*		MODE MANAGEMENT		*)
(*--------------------------*)

	IF (xEnable = TRUE) THEN
		(*	Log enabled		*)
		
		IF (xStop = TRUE) THEN
			(*	Log stopped	*)
			eEventLogMode := Log_Mode_t#Stop;
		ELSE
			(*	Otherwise log is running	*)
			eEventLogMode := Log_Mode_t#Running;
		END_IF;
		
	ELSE
		(*	Log disabled	*)
		eEventLogMode := Log_Mode_t#Disabled;
	END_IF;



	CASE eEventLogMode OF

		(*----------------------*)	
		(*		DISABLED		*)
		(*----------------------*)	
		Log_Mode_t#Disabled:
		
			(* First time that is on Disabled mode	*)
			IF (eEventLogMode_Old <> Log_Mode_t#Disabled) THEN
			
				eEventLogState := Log_State_t#Log_Disabled;
				eLogStatus := Log_Status_t#Log_Disabled;
			END_IF;


		(*----------------------*)	
		(*		RUNNING			*)
		(*----------------------*)	
		Log_Mode_t#Running:
		
			(* First time that is on Running mode	*)
			IF (eEventLogMode_Old <> Log_Mode_t#Running) THEN	
						
				eEventLogState := Log_State_t#Log_Init;
				eLogStatus := Log_Status_t#Log_Init;

				eEventLogMode_Old := eEventLogMode;				
				RETURN;	(*	The time to create a file is very long, so it exits to report that it is in the Init phase	*)
			END_IF;
			
			
		(*----------------------*)	
		(*		STOP			*)
		(*----------------------*)	
		Log_Mode_t#Stop:
		
			(* First time that is on Stop mode	*)
			IF (eEventLogMode_Old <> Log_Mode_t#Running) THEN				

 				eEventLogState := Log_State_t#Log_Disabled;
				eLogStatus := Log_Status_t#Log_Disabled;
			END_IF;

		(*----------------------*)	
		(*	SOMETHING WRONG		*)
		(*----------------------*)			
		ELSE
		
			eEventLogState := Log_State_t#Log_Disabled;
			eLogStatus := Log_Status_t#Log_Disabled;
	END_CASE;
		
	eEventLogMode_Old := eEventLogMode;


(*----------------------------------*)
(*		EVENT LOG MANAGEMENT		*)
(*----------------------------------*)

(*	An "CASE" is not used to be able to activate the log in one loop	*)

	(*----------------------*)	
	(*	INITIALIZE			*)
	(*----------------------*)			

	IF (eEventLogState = Log_State_t#Log_Init) THEN
		eEventLogState := Log_State_t#Log_Check;
	END_IF;

			
	(*------------------------------*)	
	(*	CHECK THE EVENT LOG FILE	*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_Check) THEN

		strFile := CONCAT(STR_EVENTLOG_FILE_PATH, '/');
		strFile := CONCAT(strFile, STR_EVENTLOG_FILE_NAME);
		strFile := CONCAT(strFile, STR_EVENTLOG_FILE_EXTENSION);

		usiRet := clsEventLog.FilePresent(	xMemoryFlashPublic,		(*	xMemoryFlashPublic				*)
											strFile);				(*	strFilePathAndNameWithExtension	*)
		
		eLogError := Log_Error_t#LogSuccess;
		eLogStatus := Log_Status_t#Log_Init;
		
		CASE usiRet OF
			
			1: 	(*	File not present		*)
			
				(*	Create a new log		*)
				eEventLogState := Log_State_t#Log_CreateNew;
				
			2:	(*	Check the header		*)
			
				(*	check the header		*)
				eEventLogState := Log_State_t#Log_CheckHeader;
			
			ELSE	(*	0: Error on load the file	*)
		
				eEventLogState := Log_State_t#Log_Error;
				eLogError := Log_Error_t#LogError_LoadConfig;
		
		END_CASE;
		
	END_IF;


	(*------------------------------*)	
	(*	CHECK THE EVENT LOG HEADER	*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_CheckHeader) THEN


		usiRet := TO_USINT(clsEventLog.GetHeader(	ADR(a_usiDataBuff[1]),		(*	pnt_usiBuffer	*)
								   					EVENTLOG_DATASIZE ));		(*	uiBufferSize	*)

		IF (usiRet = 0) THEN
			(*	Error on load the header	*)
			
			eLogStatus := Log_Status_t#Log_Error;																							
			eEventLogState := Log_State_t#Log_Error;	
			eLogError := Log_Error_t#LogError_FileHeader;		
		ELSE
		
			xDummy := FALSE;
			
			IF (usiRet = EVENTLOG_HEADERSIZE) THEN
				(*	Header size as expected: check the header	*)
				
				xDummy := TRUE;
	
				IF (TO_USINT(udiLogAppType) <> a_usiDataBuff[1]) THEN
					xDummy := FALSE;
				END_IF;
				
				IF (TO_USINT(SHR(udiLogAppType,8)) <> a_usiDataBuff[2]) THEN
					xDummy := FALSE;
				END_IF;
				
				IF (TO_USINT(SHR(udiLogAppType,16)) <> a_usiDataBuff[3]) THEN
					xDummy := FALSE;
				END_IF;
				
				IF (TO_USINT(SHR(udiLogAppType,24)) <> a_usiDataBuff[4]) THEN
					xDummy := FALSE;
				END_IF;
				
				IF (TO_USINT(uiLogSWVersion) <> a_usiDataBuff[5]) THEN
					xDummy := FALSE;
				END_IF;
				
				IF (TO_USINT(SHR(uiLogSWVersion,8)) <> a_usiDataBuff[6]) THEN
					xDummy := FALSE;
				END_IF;
				
			END_IF;
			
			(*	Check the records number	*)
			IF (xDummy = TRUE) THEN
				
				udiAux := clsEventLog.GetMaxNumRecords();
									
				IF (udiAux <> udiNumRecords) THEN
					xDummy := FALSE;
				END_IF;
			END_IF;			
			
			IF (xDummy = TRUE) THEN
				(*	Header/Size OK		*)
						
				eEventLogState := Log_State_t#Log_Recording;		
				eLogStatus := Log_Status_t#Log_Recording;
				eLogError := Log_Error_t#LogSuccess;							
			ELSE
				(*	Headers/Size are not equal: save current log and create a new one	*)

				xDummy := clsEventLog.ReleaseHandler();

				fbDateAndTime();
					
				strFile := CONCAT(STR_EVENTLOG_FILE_PATH, '/');
				strFile := CONCAT(strFile, STR_EVENTLOG_FILE_NAME);

				xDummy := LogRenameFile(xMemoryFlashPublic,				(*	xMemoryFlashPublic	*)
										strFile,						(*	strFilePathAndName	*)
										STR_EVENTLOG_FILE_EXTENSION,	(*	Logger extension	*)
										TRUE,							(*	xAddDate			*)
										fbDateAndTime.uiYear,			(*	Year				*)
										fbDateAndTime.usiMonth,			(*	Month				*)
										fbDateAndTime.usiDay);			(*	Day					*)

				IF (xDummy = FALSE) THEN
					(*	Something goes wrong	*)	
					eEventLogState := Log_State_t#Log_Error;
					eLogStatus := Log_Status_t#Log_Init;	
					eLogError := Log_Error_t#LogError_CopyFile;
				ELSE
					(*	Header KO: generate a new file	*)
					eEventLogState := Log_State_t#Log_CreateNew;	
					eLogStatus := Log_Status_t#Log_Init;	
					eLogError := Log_Error_t#LogSuccess;																
				END_IF;
			END_IF;		
		END_IF;		
	END_IF;
	
	
	(*------------------------------*)	
	(*	CREATE A NEW EVENT LOG FILE	*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_CreateNew) THEN

		strFile := CONCAT(STR_EVENTLOG_FILE_NAME, STR_EVENTLOG_FILE_EXTENSION);

		a_usiDataBuff[1] := TO_USINT(udiLogAppType);	
		a_usiDataBuff[2] := TO_USINT(SHR(udiLogAppType,8));					
		a_usiDataBuff[3] := TO_USINT(SHR(udiLogAppType,16));					
		a_usiDataBuff[4] := TO_USINT(SHR(udiLogAppType,24));					
		a_usiDataBuff[5] := TO_USINT(uiLogSWVersion);					
		a_usiDataBuff[6] := TO_USINT(SHR(uiLogSWVersion,8));	


		usiRet := clsEventLog.NewFile(	xMemoryFlashPublic,				(*	xMemoryFlashPublic			*)
										STR_EVENTLOG_FILE_PATH,			(*	strFilePath					*)
									  	strFile,						(*	strFileNameWithExtansion	*)
									  	udiNumRecords,					(*	udiNumOfRecords				*)
									  	EVENTLOG_BUFFERNUMBEROFRECORDS,	(*	uiBuffNumOfRecords			*)
									  	EVENTLOG_HEADERSIZE,			(*	uiHeaderSize				*)
									  	EVENTLOG_DATASIZE,				(*	uiDataSize					*)
									  	TRUE,							(*	xCircular					*)
									  	EVENTLOG_DESCRIPTION,			(*	strDescription				*)
									  	EVENTLOG_HEADERSIZE,			(*	uiBuffHeaderSize			*)
									  	ADR(a_usiDataBuff[1]));			(*	pBuffHeader					*)
										
		IF (usiRet = 0) THEN
			(*	Error condition	*)

			eEventLogState := Log_State_t#Log_Error;
			eLogError := Log_Error_t#LogError_CreateNewLog;
			eLogStatus := Log_Status_t#Log_Error;
		ELSE
			(*	New file OK		*)

			eEventLogState := Log_State_t#Log_Recording;
			eLogError := Log_Error_t#LogSuccess;
			eLogStatus := Log_Status_t#Log_Recording;
		END_IF;

	END_IF;


	(*------------------------------*)	
	(*	RECORD A NEW EVENT			*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_Recording) THEN
		
		
		(*	Dequeue all the events	*)
		xExit := TRUE;
		
		usiAux := 0;
		usiItems := 0;
		
		WHILE ((clsQueueEventLog.Empty() = FALSE) AND (xExit = TRUE) AND (usiItems < usiQueueMaxItems)) DO
		
			usiItems := usiItems + 1;
			xDummy := clsQueueEventLog.Dequeue(ADR(eventLogData));
		
			IF (xDummy = TRUE) THEN
				
				usiAux := usiAux + 1;
				
				(*	Data ready	*)
				xDummy := clsEventLog.SetOtherEvent(REF(eventLogData),	(*	eventLog	*)
											   REF(otherEvent));	(*	otherEvent	*)
				
				usiRet := clsEventLog.AddRecord(EVENTLOG_DATASIZE,	(*	uiBuffSize	*)
										   ADR(otherEvent));	(*	pBuff		*)
					
				CASE usiRet OF
				
				0:	(*	OK			*)
								
				1:	(*	Generic Error	*)
				
					eEventLogState := Log_State_t#Log_Error;
					eLogError := Log_Error_t#LogError_Recording;
					eLogStatus := Log_Status_t#Log_Error;				
					xExit := FALSE;
				
				2:	(*	File full	*)
				
					eEventLogState := Log_State_t#Log_Error;
					eLogError := Log_Error_t#LogError_Full;
					eLogStatus := Log_Status_t#Log_Error;				
					xExit := FALSE;
					
				3: 	(*	Time not supported	*)
				
					eEventLogState := Log_State_t#Log_Error;
					eLogError := Log_Error_t#LogError_TimeNotSupported;
					eLogStatus := Log_Status_t#Log_Error;				

				END_CASE;
										  			
			END_IF;
			
		END_WHILE;
		
		(*	Refresh the max number of items found into the queue	*)
		IF (usiAux > usiQueueMaxItems) THEN		
			usiQueueMaxItems := usiAux;
		END_IF;
		
	END_IF;
	

	(*------------------------------*)	
	(*	EVENT LOG DISABLED			*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_Disabled) THEN
		(*	Release the Event Log file handler	*)
			
		xDummy := clsEventLog.ReleaseHandler();
			
		eEventLogState := Log_State_t#Log_Disabled;
		eLogError := Log_Error_t#LogSuccess;
		eLogStatus := Log_Status_t#Log_Disabled;
	END_IF;
	
	
	(*------------------------------*)	
	(*	EVENT LOG ERROR				*)
	(*------------------------------*)			
	IF (eEventLogState = Log_State_t#Log_Error) THEN

		(*	Release the Event Log file handler	*)
		xDummy := clsEventLog.ReleaseHandler();

		eLogStatus := Log_Status_t#Log_Error;
	END_IF;

{ ENDIF }
