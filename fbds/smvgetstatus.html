SMVGetStatus

Valve Control Intermediate FunctionBlock v1.0.0

Description

Returns the operating status of the stepper valve.

Inputs (2)

Name Type Init Description
device Device_t Device_t. Device ID
peripheral Peripheral_t Peripheral_t. Peripheral ID

Outputs (8)

Name Type Description
xEnabled BOOL BOOL, range [FALSE, TRUE]. TRUE: enable the valve
uiCurrentPerc UINT UINT, range [0,10000], unit [0.01%]. Current stepper valve position
udiCurrentUSteps UDINT UDINT. Current stepper valve micro steps
udiCurrentFullSteps UDINT
eStatus StepperDriverStatus_t Current stepper valve status
xForceOverdrive BOOL BOOL, range [FALSE, TRUE]. TRUE: force overdrive
eVirtualIIO_Status VirtualIO_Status_t Current IOManager stepper valve status
usiValveTypeIndex USINT Valve Type Index

Local Variables (1)

Name Type Description
usiIndex USINT

Source Code (ST)

xEnabled := FALSE;
eVirtualIIO_Status := VirtualIO_Status_t#NotConfigured;

FOR usiIndex := 0 TO CONST_IOSMVLastIndex DO

	IF ((SMVDriver[usiIndex].device = device) AND
		(SMVDriver[usiIndex].peripheral = peripheral)) THEN

		xEnabled := SMVDriver[usiIndex].xEnabling;
		uiCurrentPerc := SMVDriver[usiIndex].uiCurrentPerc;
		udiCurrentUSteps := SMVDriver[usiIndex].udiCurrentUSteps;

		CASE SMVDriver[usiIndex].usiSteppingMode OF

		0: (* 0=Full *)
				
			udiCurrentFullSteps := udiCurrentUSteps;
		
		1: (* 1=Half *)

			udiCurrentFullSteps := udiCurrentUSteps / 2;
			
		2: (* 2=1/4 *)

			udiCurrentFullSteps := udiCurrentUSteps / 4;
			
		3: (* 3=1/8 *)

			udiCurrentFullSteps := udiCurrentUSteps / 8;
			
		4: (* 4=1/16 *)

			udiCurrentFullSteps := udiCurrentUSteps / 16;

		5: (* 5=1/32 *)

			udiCurrentFullSteps := udiCurrentUSteps / 32;

		ELSE
												
			udiCurrentFullSteps := udiCurrentUSteps;

		END_CASE;		
				
				
		eStatus := SMVDriver[usiIndex].eStatus;
		xForceOverdrive := SMVDriver[usiIndex].xForceOverdrive;
		eVirtualIIO_Status := SMVDriver[usiIndex].eVirtualIIO_Status;
	
		usiValveTypeIndex := SMVDriver[usiIndex].usiValveTypeIndex;
	
		RETURN;		
		
	END_IF;

END_FOR;
