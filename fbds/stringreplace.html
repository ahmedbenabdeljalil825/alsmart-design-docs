//	Default: nothing inserted
	StringReplace := 0;

	//	If the search string is null -> Exit
	IF (uiStrFindSize = 0) THEN
		RETURN;
	END_IF;

	pByteSource := TO_POINTER(pStrSource);
	pByteFind := TO_POINTER(pStrFind);
	uiSourcePosition := 0;
	uiFindPosition := 0;
	xLoop := TRUE;


	REPEAT

		IF (@pByteSource = @pByteFind) THEN
	
			// Same char
			pByteFind := pByteFind + 1;
			uiFindPosition := uiFindPosition + 1;
				
			IF (uiFindPosition = uiStrFindSize) THEN
				// Search string found -> replace the string
			
				IF (uiStrReplaceSize > uiStrFindSize) THEN
					// String to insert larger than the removed string
					// Check if the new size is acceptable
					IF ((@pStrSourceSize + uiStrReplaceSize	- uiStrFindSize) > uiStrSourceMaxSize) THEN
						// New size too big					
						StringReplace := -1;
						RETURN;
					END_IF;			
				
					// Move the remainder forward
					pByteFrom := pByteSource;
					
					FOR uiIndex := (uiSourcePosition + 1) TO (@pStrSourceSize) DO									
						pByteFrom := pByteFrom + 1;					
					END_FOR;

					pByteTo := pByteFrom;
						
					FOR uiIndex := 1 TO (uiStrReplaceSize - uiStrFindSize ) DO									
						pByteTo := pByteTo + 1;					
					END_FOR;
	
					FOR uiIndex := (uiSourcePosition + 1) TO (@pStrSourceSize) DO					
						@pByteTo := @pByteFrom;					
						pByteFrom := pByteFrom - 1;
						pByteTo := pByteTo - 1;						
					END_FOR;
				
				ELSE
					IF (uiStrReplaceSize < uiStrFindSize) THEN
						// String to insert smaller than the removed string	
						// Move the remainder part back				
						pByteFrom := pByteSource + 1;
						pByteTo := pByteSource;
					
						FOR uiIndex := 2 TO (uiStrFindSize - uiStrReplaceSize) DO					
							pByteTo := pByteTo - 1;					
						END_FOR;

						FOR uiIndex := (uiSourcePosition + 1) TO (@pStrSourceSize) DO					
							@pByteTo := @pByteFrom;					
							pByteFrom := pByteFrom + 1;
							pByteTo := pByteTo + 1;						
						END_FOR;

					ELSE
						// String to insert equal to the removed string						
						// Do nothing
					END_IF;		
		
				END_IF;
			
				// Insert the new string
				@pStrSourceSize := @pStrSourceSize + uiStrReplaceSize - uiStrFindSize;

				FOR uiIndex := 1 TO (uiStrFindSize - 1) DO					
					pByteSource := pByteSource - 1;
					uiSourcePosition := uiSourcePosition - 1;					
				END_FOR;					
					
				pByteReplace := TO_POINTER(pStrReplace);
					
				FOR uiIndex := 1 TO uiStrReplaceSize DO					
					@pByteSource := @pByteReplace;
					pByteReplace := pByteReplace + 1;
					pByteSource := pByteSource + 1;	
					uiSourcePosition := uiSourcePosition + 1;										
				END_FOR;
			
				StringReplace := StringReplace + 1;
		
			END_IF;
		
		ELSE
			// Different char -> re-init the serach
			pByteFind := TO_POINTER(pStrFind);
			uiFindPosition := 0;
		END_IF;

		pByteSource := pByteSource + 1;
		uiSourcePosition := uiSourcePosition + 1;
	
	UNTIL

		((xLoop = FALSE) OR (uiSourcePosition >= @pStrSourceSize))

	END_REPEAT;
